484 // Reads count bytes into p, deferring to mpack_read_native_big() if more
485 // bytes are needed than are available in the buffer.
486 MPACK_INLINE_SPEED void mpack_read_native(mpack_reader_t* reader, char* p, size_t count);
487 
488 #if MPACK_DEFINE_INLINE_SPEED
489 MPACK_INLINE_SPEED void mpack_read_native(mpack_reader_t* reader, char* p, size_t count) {
490  if (count > reader->left) {
491  mpack_read_native_big(reader, p, count);
492  } else {
493  mpack_memcpy(p, reader->buffer + reader->pos, count);
494  reader->pos += count;
495  reader->left -= count;
496  }
497 }
498 #endif

514 MPACK_ALWAYS_INLINE uint8_t mpack_read_native_u8(mpack_reader_t* reader) {
515  if (reader->left >= sizeof(uint8_t)) {
516  uint8_t ret = mpack_load_native_u8(reader->buffer + reader->pos);
517  reader->pos += sizeof(uint8_t);
518  reader->left -= sizeof(uint8_t);
519  return ret;
520  }
521 
522  char c[sizeof(uint8_t)];
523  mpack_read_native_big(reader, c, sizeof(c));
524  return mpack_load_native_u8(c);
525 }
