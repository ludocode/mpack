518 // bytes are needed than are available in the buffer.
519 MPACK_INLINE_SPEED void mpack_read_native(mpack_reader_t* reader, char* p, size_t count);
520 
521 #if MPACK_DEFINE_INLINE_SPEED
522 MPACK_INLINE_SPEED void mpack_read_native(mpack_reader_t* reader, char* p, size_t count) {
523  if (count > reader->left) {
524  mpack_read_native_big(reader, p, count);
525  } else {
526  mpack_memcpy(p, reader->buffer + reader->pos, count);
527  reader->pos += count;
528  reader->left -= count;
529  }
530 }
531 #endif

539 MPACK_INLINE_SPEED void mpack_read_native_nojump(mpack_reader_t* reader, char* p, size_t count) {
540  mpack_assert(reader->error == mpack_ok, "cannot call nojump if an error is already flagged!");
541  mpack_reader_error_t error_fn = reader->error_fn;

548 MPACK_ALWAYS_INLINE uint8_t mpack_read_native_u8(mpack_reader_t* reader) {
549  if (reader->left >= sizeof(uint8_t)) {
550  uint8_t ret = mpack_load_native_u8(reader->buffer + reader->pos);
551  reader->pos += sizeof(uint8_t);
552  reader->left -= sizeof(uint8_t);
553  return ret;
554  }
